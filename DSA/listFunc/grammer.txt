lit := 
    | List [expression]
    | Real

exp :=
    | literal
    | fcall

fname :=
    | id

fcall :=
    | fname [exp] 

pexp :=
    | exp
    | #int
    | fname [pexp]

fdec :=
    | fname -> pexp
